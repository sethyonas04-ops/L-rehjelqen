import openai from langchain.vectorstores import Chroma from langchain.embeddings import OpenAIEmbeddings from langchain.chains import RetrievalQA # 1. Last inn faglige kilder (f.eks. PDF-er fra universitetet eller tekniske manualer) # Vi simulerer at vi har en "hjerne" full av fagkunnskap her faglig_database = Chroma( persist_directory="./fag_kunnskap", embedding_function=OpenAIEmbeddings() ) # 2. Sett opp motoren med "Strenge Instrukser" # Vi tvinger den til å være akademisk og kildekritisk system_instruks = """ Du er en faglig ekspert-motor. Dine regler: 1. Svar KUN basert på verifiserte fakta i dokumentene. 2. Hvis du ikke finner svaret i kildene, si 'Jeg har ikke faglig grunnlag for å svare på dette'. 3. Bruk presis terminologi og oppgi kilder der det er mulig. 4. Ikke gjett. """ # 3. Lag selve "Spørre-motoren" fag_motor = RetrievalQA.from_chain_type( llm=openai.ChatCompletion(model="gpt-4o", temperature=0.1), # Lav temperatur = høyere presisjon chain_type="stuff", retriever=fag_database.as_retriever() ) # 4. Bruk motoren sporsmal = "Hva er de termodynamiske lovene i dette systemet?" svar = fag_motor.run(sporsmal) print(f"Faglig svar: {svar}")
